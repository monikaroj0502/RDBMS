DECLARE
    num1 int;
    num2 int;
    add int;
    sub int;
    multi int;
    div float;

BEGIN
     num1:=30;
     num2:=20;
    
      add:=num1+num2;
      sub:=num1-num2;
      multi:=num1*num2;
      div:=num1/num2;

DBMS_OUTPUT.PUT_LINE('addition='||add);

DBMS_OUTPUT.PUT_LINE('substraction='||sub);

DBMS_OUTPUT.PUT_LINE('multiplication='||multi);

DBMS_OUTPUT.PUT_LINE('divide='||div);


END;
